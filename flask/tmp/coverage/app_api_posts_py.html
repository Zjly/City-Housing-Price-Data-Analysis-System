9@bp.route('/posts/', methods=['POST'])

13 data = request.get_json() 

14 if not data:

16 message = {} 

17 if 'title' not in data or not data.get('title'): 

18 message['title'] = 'Title is required.' 

19 elif len(data.get('title')) > 255: 

20 message['title'] = 'Title must less than 255 characters.' 

21 if 'body' not in data or not data.get('body'): 

22 message['body'] = 'Body is required.' 

23 if message: 

24 return bad_request(message)

26 post = Post()
